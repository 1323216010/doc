### 切片

```go
package main

import "fmt"

func main() {
	arr := [5]int{1, 2, 3, 4, 5}
   s := arr[1:3]
   fmt.Println("length:", s[0:4])
   fmt.Println("length:", s)
   fmt.Println("length:", len(s))
   fmt.Println("capacity:", cap(s))
}
```
